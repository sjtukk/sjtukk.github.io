## 数据结构

### 链表

- 反转链表

  三个变量记录前中后三个位置

  ```c++
  struct Node{
      int val;
      Node* next;
      Node(int val, Node* next):val(val), next(next){};
  };
  
  Node* reverseList(Node* head){
      Node*before = nullptr;
      Node* now = head;
      while(now){
          Node* after = now->next;
          now->next = before;
          before = now;
          now = after;
      }
      return before;
  }
  ```

- 合并升序链表

  ```c++
  struct Node{
      int val;
      Node* next;
      Node(int val, Node* next):val(val), next(next){};
      Node(int val):val(val), next(nullptr){};
  };
  
  Node* merge(Node* head1, Node* head2){
      Node* head = nullptr;
      Node* tail = nullptr;
      while(head1 || head2){
          int n1 = head1? head1->val: INT_MAX;
          int n2 = head2? head2->val: INT_MAX;
          int val = n1 < n2 ? n1:n2;
          if(!head){
              head = tail = new Node(val);
          }else{
              tail -> next = new Node(val);
              tail = tail -> next;
          }
      }
  }
  ```

## 排序

- 快排

  ```c++
  int find_pos(vector<int>&vec, int low, int high){
      int pivot = vec[low];
      int i = low + 1;
      int j = high;
      while(true){
          while(vec[i] <= pivot) i++;
          while(vec[j] >= pivot) j--;
          if(i >= j){
              break;
          }
          swap(vec[i], vec[j]);
      }
      swap(vec[low], vec[j]);
      return j;
  }
  
  void quickSort(vector<int>&vec, int low, int high){
      if(low < high){
          int pos = find_pos(vec, low, high);
          quickSort(vec, low, pos - 1);
          quickSort(vec, pos + 1, high);
      }
  }
  
  int main(){
      vector<int>vec = {1,23,4,6,9,-5};
      int n = vec.size();
      quickSort(vec, 0, n - 1);
  }
  ```

  ```c++
          p = p->next;
          q = q->next->next;

      p->next = nullptr;
      return merge(sortList(head), sortList(head2));
  ```

## 算法



- 如何找到100亿个数字中的top100

  - 小根堆

    开始建一个大小为100的小根堆，然后遍历100亿个数字，如果比根节点还要小，不插入，如果比根节点大，那么进行插入，并删除根节点

  - hash

    如果数字中有很多重复的，可以先用hash去重，然后再用其他方法找top100。但是会出现hash碰撞的问题

  - 多路归并

    把数据分成多份，每份求top100，然后进行多路归并
    
    

- 最长公共子序列(the longest common subsequece)

  dp[i] [j] 的含义是：s1从[0, i]，s2从[0,j]，这两个字符串含义的最长公共子序列的长度

  ```c++
  // 递归写法 top - down
  int func(string& s1, string& s2, int m, int n){
      if(m < 0 || n < 0) return 0;
      if(s1[m] == s2[n]) return 1 + func(s1, s2, m - 1, n - 1);
      return max(func(s1, s2, m - 1, n), func(s1, s2, m, n - 1));
  }
  
  // 动态规划 down - top
  int func1(string& s1, string& s2){
      int n1 = s1.size();
      int n2 = s2.size();
      vector<vector<int>> dp(n1 + 1, vector<int>(n2 + 1, 0));
      for(int i = 1; i <= n1; i++){
          for(int j = 1; j <= n2; j++){
              if(s1[i] == s2[j]) dp[i][j] = dp[i - 1][j - 1] + 1;
              else dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
          }
      }
      return dp[n1][n2];
  }
  
  int main(){
      string s1 = "AGGTAB";
      string s2 = "GXTXAYB";
      cout << func1(s1, s2);
  }
  ```

- 最长公共子串

  最长公共子序列的变体(the longest common substring)

  dp[i] [j] 的含义是：s1从[0, i]，s2从[0,j]，以s1[i]结尾的最长公共子串的长度

  ```C++
  // 动态规划 down - top
  int func1(string& s1, string& s2){
      int n1 = s1.size();
      int n2 = s2.size();
      vector<vector<int>> dp(n1 + 1, vector<int>(n2 + 1, 0));
      for(int i = 1; i <= n1; i++){
          for(int j = 1; j <= n2; j++){
              if(s1[i] == s2[j]) dp[i][j] = dp[i - 1][j - 1] + 1;
              else dp[i][j] = 0;
          }
      }
      return dp[n1][n2];
  }
  
  int main(){
      string s1 = "AGGTAB";
      string s2 = "GXTXAYB";
      cout << func1(s1, s2);
  }
  ```
